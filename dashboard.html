// 🔔 효과음 및 음성 안내(TTS) 제어 함수 function notify(d) { // 1. 팝업창 데이터 바인딩 및 화면 표시 로직 (기존 코드 유지) document.getElementById('notiTeacher').innerText = d.teacher + " 선생님"; document.getElementById('notiStudent').innerText = d.id + " " + d.name; document.getElementById('notiPurpose').innerText = "목적: " + d.purpose; const imgEl = document.getElementById('notiImg'); if (d.imageUrl && d.imageUrl.startsWith("http")) { imgEl.src = d.imageUrl; imgEl.style.display = "block"; } else { imgEl.style.display = "none"; } document.getElementById('popup').classList.add('active'); // 🎯 2. [띵동 효과음 추가] 오디오 객체 생성 및 재생 // 고품질 오디오 리소스를 활용한 알림음 세팅 const alertSound = new Audio("https://actions.google.com/sounds/v1/alarms/digital_watch_alarm_long.ogg"); alertSound.volume = 0.5; // 소리 크기 조절 (0.0 ~ 1.0) alertSound.play().then(() => { // 3. 효과음이 끝나거나 재생된 직후(약 0.8초 뒤) TTS 음성 안내 가동 setTimeout(() => { if ('speechSynthesis' in window) { // 기존 TTS 인스턴스 초기화 (음성 중첩 방지) window.speechSynthesis.cancel(); const text = `${d.teacher} 선생님, ${d.id} ${d.name} 학생이 호출하였습니다.`; const speech = new SpeechSynthesisUtterance(text); speech.lang = 'ko-KR'; speech.rate = 1.0; // 말하는 속도 speech.pitch = 1.0; // 음높이 window.speechSynthesis.speak(speech); } }, 800); // 800ms(0.8초)의 마진을 두어 효과음과 음성이 겹치지 않게 조율 }).catch(err => { console.log("브라우저 오디오 상호작용 제한 발생:", err); // 만약 효과음 재생이 브라우저 정책에 막히면 TTS만이라도 즉시 실행되도록 예외 처리 if ('speechSynthesis' in window) { const text = `${d.teacher} 선생님, ${d.id} ${d.name} 학생이 호출하였습니다.`; const speech = new SpeechSynthesisUtterance(text); speech.lang = 'ko-KR'; window.speechSynthesis.speak(speech); } }); }
